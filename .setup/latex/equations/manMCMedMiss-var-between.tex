\begin{equation}
    \label{eq:mi-var-between}
    \mathbf{V}_{\mathrm{between}}
    =
    \left( 
    M - 1 
    \right)^{-1} 
    \sum_{m = 1}^{M} 
    \left( 
    \hat{\boldsymbol{\theta}}_{m}
    - 
    \bar{\boldsymbol{\theta}} 
    \right) 
    \left( 
    \hat{\boldsymbol{\theta}}_{m}
    - 
    \bar{\boldsymbol{\theta}} 
    \right)^{\prime}
\end{equation}
