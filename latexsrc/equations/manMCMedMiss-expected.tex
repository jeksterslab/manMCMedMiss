\begin{equation}
	\label{eq:expected}
	\mathbb{E}
	\left[
	\begin{array}{c}
		{X} \\
		{M} \\
		{Y} 
	\end{array}
	\right]
	=
	\left[
	\begin{array}{c}
		{\mu}_{X} \\
		{\mu}_{M} \\
		{\mu}_{Y} 
	\end{array}
	\right]
	=
	\left[
	\begin{array}{c}
		{\mu}_{X}                  \\
		{\delta}_{M}               
		+ {\alpha}{\mu}_{X}        \\
		{\delta}_{Y}               
		+ {\tau}^{\prime}{\mu}_{X} 
		+ {\beta}{\mu}_{M}         
	\end{array}
	\right] .
\end{equation}
